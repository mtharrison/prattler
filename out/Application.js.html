var Room = require('./Room')
  , User = require('./User')

module.exports = Application

/**
 * The chat application
 * @constructor
 */
function Application() {
    this.rooms = {}
    this.users = {}
}

/**
 * Checks if the room currently exists
 *
 * @param {string} name - the name/identifier of the room
 */
Application.prototype.roomExists = function(name) {
    return this.rooms.hasOwnProperty(name);
}

/**
 * Adds a user to the user pool
 *
 * @param {object} user - options for the user (name, id)
 */
Application.prototype.addUser = function(user) {
    this.users[user.id] = user
}

/**
 * Creates or retrieves a room. First checks if the room exists
 * from the name and returns the Room object if so
 *
 * @param {object} roomOptions - options for the room
 * @return {Room} The actual room object
 */
Application.prototype.createRoom = function(roomOptions) {
    var room = this.roomExists(roomOptions.name) 
               ? this.rooms[roomOptions.name] 
               : new Room(roomOptions)

    this.rooms[room.name] = room
    return room
}
